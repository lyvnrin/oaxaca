Module: App

Root component that sets up the application routing structure. Provides navigation between landing pages, authentication pages, staff dashboards, and the customer menu interface.
Source:
  • App.jsx, line 6
Returns:
The application with configured routes
Type
JSX.Element
Examples
// Usage in main.jsx:
import React from 'react';
import ReactDOM from 'react-dom/client';
import App from './App';

ReactDOM.createRoot(document.getElementById('root')).render(
  <React.StrictMode>
    <App />
  </React.StrictMode>
);
// Route structure:
// - Landing pages: "/" (public landing), "/staff" (staff role selection)
// - Authentication: "/customer-login", "/waiter-login", "/kitchen-login", "/manager-login"
// - Staff dashboards: "/waiter-dashboard", "/kitchen-dashboard", "/manager-dashboard"
// - Customer menu: "/menu"
